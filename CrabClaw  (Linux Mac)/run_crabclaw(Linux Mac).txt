#!/usr/bin/env bash
# 商业级安全捕获与启动保护
set -e

RED='\033[0;31m'
GREEN='\033[0;32m'
YELLOW='\033[0;33m'
BLUE='\033[0;34m'
NC='\033[0m' 

echo -e "${BLUE}===================================================${NC}"
echo ""
echo -e "       🌟 CrabClaw 商业开源引擎 (Pro Max) 🌟"
echo ""
echo -e "${BLUE}===================================================${NC}"

cd "$(dirname "$0")" || { echo -e "${RED}❌ 无法切换到脚本所在目录！${NC}"; exit 1; }

OS="$(uname -s)"
echo -e "${GREEN}[系统检测]${NC} 当前平台: $OS"

if command -v python3 &>/dev/null; then
    PYTHON_CMD="python3"
elif command -v python &>/dev/null; then
    PYTHON_CMD="python"
else
    echo -e "${RED}❌ [致命错误] 未检测到 Python 运行环境！${NC}"
    exit 1
fi

echo -e "${GREEN}[系统自检]${NC} 正在验证框架依赖库..."
# 禁用 set -e 以免此处退出
set +e
$PYTHON_CMD -c "import openai, httpx" &>/dev/null
if [ $? -ne 0 ]; then
    echo -e "${YELLOW}⚠️ 发现依赖缺失，激活双源灾备策略进行极速补全...${NC}"
    PIP_CMD="$PYTHON_CMD -m pip"
    $PIP_CMD install --user openai httpx -i https://mirrors.aliyun.com/pypi/simple/ || $PIP_CMD install --user openai httpx --break-system-packages -i https://pypi.tuna.tsinghua.edu.cn/simple
    echo -e "${GREEN}✅ 补全完成！${NC}"
fi
set -e

if [ ! -f "crabclaw.py" ]; then
    echo -e "${RED}❌ [丢失警报] 找不到核心入口文件 crabclaw.py！${NC}"
    exit 1
fi

echo -e "${BLUE}===================================================${NC}"
echo -e "🚀 引擎点火中..."
echo -e "${BLUE}===================================================${NC}"

export PYTHONIOENCODING=utf-8
set +e
$PYTHON_CMD crabclaw.py
echo -e "${YELLOW}⚠️ 引擎运行结束或发生异常闪退。${NC}"
